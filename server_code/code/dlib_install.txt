# Download and make DLIB
# Source: https://stackoverflow.com/questions/51697468/how-to-check-if-dlib-is-using-gpu-or-not/51698961
# Watch for the message of whether DLIB will be made with CUDA or not.
git clone https://github.com/davisking/dlib.git
cd dlib
mkdir build
cd build
cmake .. -DDLIB_USE_CUDA=1 -DUSE_AVX_INSTRUCTIONS=1
cmake --build .
cd ..
python setup.py install --yes USE_AVX_INSTRUCTIONS --yes DLIB_USE_CUDA


## Overall success is defined with opening python, importing dlib, and running:
dlib.DLIB_USE_CUDA ## Should equal True.
